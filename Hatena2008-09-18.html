    lines = [[]]+[[x]for x in lines]
    N -= True

def mul(xs, n, result=""):
    if xs:
        return mul(xs.pop()[:], n, result + n)
    return result

def to_num(xs, result=False):
    if xs:
        return to_num(xs.pop()[:], result + True)
    return result

    for p in primes:

    else:
        primes.append(i)

print [to_num(p[:]) for p in primes]
