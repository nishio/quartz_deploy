def blute(N, M):
    """
    >>> blute(2, 2)
    2

    >>> blute(2, 3)
    18
    """
    from itertools import permutations
    nums = range(M)
    count = 0
    for a in permutations(nums, N):
        for b in permutations(nums, N):
            if all(a[i] != b[i] for i in range(N)):
                count += 1
    print(count)
