>|python|
>>> def Struct(*keys):
	class _Struct(object):
	    def __init__(self, *values):
		self.__dict__.update(zip(keys, values))
	return _Struct

	
>>> Struct("foo", "bar")
<class '__main__._Struct'>
>>> FooBar = Struct("foo", "bar")
>>> FooBar(1, 2)
<__main__._Struct object at 0x01494E90>

1
