            try:
                while True:
                    print "getting data",

                    print "got data, len", len(data)
                    if len(data) == 0:
                        TO_CLOSE = True
                        break
                    print "sending data",
                    s.send(data)
                    print "sent data"

                print repr(e)
                
            try:
                while True:
                    print "getting result",
                    result = s.recv(8192)

                    if "#include" in result:
