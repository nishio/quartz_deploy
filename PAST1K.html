def solve(N, PS, Q, QS):

    # to 0-origin
    PS = [p - 1 for p in PS]
    from collections import defaultdict
    children = defaultdict(list)
    for i, p in enumerate(PS):
        children[p].append(i)

    # calc depth
    depth = [0] * N

    def visit(v):
        d = depth[v] + 1
        for c in children[v]:
            depth[c] = d
            visit(c)

    root = children[-2][0]
    visit(root)

        next = [0] * N

            next[i] = prev[prev[i]]
        parents.append(next)

        a -= 1

        d = depth[a] - depth[b]
