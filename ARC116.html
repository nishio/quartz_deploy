def solve(N, M):
    if M % 2:
        return 0
    M //= 2
    MOD = 998_244_353
    comb = CombinationTable(N + 10, MOD).comb

    cache = {}    
    def foo(x, start=11):
        if (x, start) in cache:
            return cache[(x, start)]
        if start == 0:
            if 2 * x > N:
                return 0
            else:
                ret = comb(N, 2 * x)
                cache[(x, start)] = ret
                return ret

        p = 2 ** start
        if p <= x:
            ret = 0
            for i in range(x // p + 1):
                ret = (ret + foo(x - i * p, start - 1) * comb(N, 2 * i)) % MOD
        else:
            ret = foo(x, start - 1)
        cache[(x, start)] = ret
        return ret

    return foo(M)
