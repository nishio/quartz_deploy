It was pointed out to me that the first code wasn't tail recursion, so I rewrote it and it became O(n ^ 2).

>|haskell|
import System

pow :: Integer -> Integer -> Integer
pow x 0 = x
pow x n = pow (7 * x) (n - 1)

main = do
  args <- getArgs
  print $ (0 *) $ pow 1 $ read $ args !! 0
