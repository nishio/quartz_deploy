def solve(N, K):
    def f(x):
        if x < 2:
            return 0
        if x > 2 * N:
            return 0
        return N - abs(x - (N + 1))

    ret = 0
    for ab in range(2, 2 * N + 1):
        cd = ab - K
        ret += f(ab) * f(cd)

    return ret
