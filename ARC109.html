    ret = x

        a -= 1

    return ret

def solve(N, K, S):
    last = [0 if c == "R" else 1 if c == "P" else 2 for c in S]
    match = [0, 1, 0, 1, 1, 2, 0, 2, 2]

    for _k in range(K - 1, -1, -1):
        next = []
        for i in range(0, 2 * N, 2):
            next.append(match[last[i % N] * 3 + last[(i + 1) % N]])
        last = next
    return "RPS"[last[0]]
